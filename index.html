  <title>マナビバ入退室</title>

  <script src="https://unpkg.com/html5-qrcode"></script>

  <style>

    body { font-family: sans-serif; text-align: center; margin: 0; padding: 20px; background: #f0f2f5; }

    #reader { width: 100%; max-width: 500px; margin: 0 auto; border-radius: 10px; overflow: hidden; }

    #result { margin-top: 20px; font-size: 1.2em; font-weight: bold; padding: 15px; border-radius: 8px; }

    .loading { color: #666; }

    .success { background: #d4edda; color: #155724; }

  </style>

</head>

<body>

  <h2>マナビバ入退室管理</h2>

  <div id="reader"></div>

  <div id="result" class="loading">カメラを起動中...</div>


  <script>

    // ★ここにあなたのGASのウェブアプリURLを貼り付けてください

    const GAS_URL = "https://script.google.com/macros/s/AKfycbxs2i-UDDR_ddUW902Z-oHHS_X133LjccSXEDzLNN143Vxi3aunK2g60VbPwNb0SzWu/exec";


    const html5QrCode = new Html5Qrcode("reader");

    const config = { fps: 15, qrbox: { width: 250, height: 250 } };


    html5QrCode.start({ facingMode: "environment" }, config, (decodedText) => {

      // 読み取り成功時

      html5QrCode.pause(); // 二重送信防止

      const resultDiv = document.getElementById('result');

      resultDiv.innerText = "通信中...";


      // GASへデータを送信

      fetch(GAS_URL, {

        method: "POST",

        mode: "no-cors", // これが重要

        headers: { "Content-Type": "application/x-www-form-urlencoded" },

        body: "id=" + encodeURIComponent(decodedText)

      })

      .then(() => {

        resultDiv.innerText = "記録完了: " + decodedText;

        resultDiv.className = "success";

        setTimeout(() => {

          resultDiv.innerText = "QRコードをかざしてください";

          resultDiv.className = "loading";

          html5QrCode.resume();

        }, 3000);

      })

      .catch(err => {

        alert("送信エラー: " + err);

        html5QrCode.resume();

      });

    });

  </script>

</body>
